def encodage(df):
    for col in df.select_dtypes('object'):
        encoder = LabelEncoder()
        encode = encoder.fit_transform(df[col].unique())
        cl=df[col].unique()
        code = {a:b for a,b in zip(cl,encode)}
        df[col] = df[col].map(code)
        #print(encode)
        #print(cl)
        #print(code)
    return df
